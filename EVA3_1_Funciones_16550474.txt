CREATE DEFINER=`root`@`localhost` FUNCTION `suma_numeros`(num1 int, num2 int) 
RETURNS int(11)

BEGIN
  
	declare suma int;
    
	set suma = num1 + num2;
    
	RETURN suma;

END

CREATE DEFINER=`root`@`localhost` FUNCTION `contar_clientes`() 
RETURNS int(11)

BEGIN
	
	declare cont int;
    
	select count(*) into cont from customers;
	
	RETURN cont;

END
